--- FILE 2: spike/run-tests.ps1 ---

Create a convenience script that runs all test cases in sequence with output suitable for copy-pasting into SPIKE-RESULTS.md:

```powershell
#Requires -Version 5.1
<#
.SYNOPSIS
    Runs all Phase 1 spike test cases and outputs results for SPIKE-RESULTS.md.
    Execute from the spike\ directory on the Windows test machine.
#>
Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

$ScriptDir = $PSScriptRoot
Write-Host "=== Phase 1 Spike Test Run ===" -ForegroundColor Cyan
Write-Host "Date: $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss')"
Write-Host ""

# --- TC-01: Left Half ---
Write-Host "--- TC-01: Left Half (X=0 Y=0 Width=960 Height=1080) ---" -ForegroundColor Yellow
& "$ScriptDir\Set-WacomMapping.ps1" -X 0 -Y 0 -Width 960 -Height 1080
Write-Host ">> Verify: stylus should be restricted to left half of display"
Write-Host ">> Press Enter to continue to TC-02..." ; Read-Host

# --- TC-02: Right Half ---
Write-Host "--- TC-02: Right Half (X=960 Y=0 Width=960 Height=1080) ---" -ForegroundColor Yellow
& "$ScriptDir\Set-WacomMapping.ps1" -X 960 -Y 0 -Width 960 -Height 1080
Write-Host ">> Verify: stylus should be restricted to right half of display"
Write-Host ">> Press Enter to continue to TC-03..." ; Read-Host

# --- TC-03: Centre Region ---
Write-Host "--- TC-03: Centre Region (X=240 Y=270 Width=1440 Height=540) ---" -ForegroundColor Yellow
& "$ScriptDir\Set-WacomMapping.ps1" -X 240 -Y 270 -Width 1440 -Height 540
Write-Host ">> Verify: stylus should be restricted to centre rectangle"
Write-Host ">> Press Enter to continue to TC-04 (latency test)..." ; Read-Host

# --- TC-04: Three Consecutive Changes for Latency ---
Write-Host "--- TC-04: Three Consecutive Changes (latency measurement) ---" -ForegroundColor Yellow
Write-Host "Latency is printed by Set-WacomMapping.ps1 for each run. Copy to SPIKE-RESULTS.md."
Write-Host ""

Write-Host "Run 1/3:"
$t1 = Measure-Command { & "$ScriptDir\Set-WacomMapping.ps1" -X 0 -Y 0 -Width 960 -Height 1080 }
Write-Host "TC-04 Run 1 total wall time: $([Math]::Round($t1.TotalMilliseconds)) ms"

Write-Host "Run 2/3:"
$t2 = Measure-Command { & "$ScriptDir\Set-WacomMapping.ps1" -X 960 -Y 0 -Width 960 -Height 1080 }
Write-Host "TC-04 Run 2 total wall time: $([Math]::Round($t2.TotalMilliseconds)) ms"

Write-Host "Run 3/3:"
$t3 = Measure-Command { & "$ScriptDir\Set-WacomMapping.ps1" -X 240 -Y 270 -Width 1440 -Height 540 }
Write-Host "TC-04 Run 3 total wall time: $([Math]::Round($t3.TotalMilliseconds)) ms"

$max = [Math]::Max($t1.TotalMilliseconds, [Math]::Max($t2.TotalMilliseconds, $t3.TotalMilliseconds))
$mean = ($t1.TotalMilliseconds + $t2.TotalMilliseconds + $t3.TotalMilliseconds) / 3
Write-Host ""
Write-Host "Mean: $([Math]::Round($mean)) ms   Max: $([Math]::Round($max)) ms"
if ($max -lt 3000) {
    Write-Host "SPIKE-02: PASS (all runs < 3000 ms)" -ForegroundColor Green
} else {
    Write-Host "SPIKE-02: FAIL (at least one run >= 3000 ms)" -ForegroundColor Red
}

Write-Host ">> Press Enter to continue to TC-05 (reset)..." ; Read-Host

# --- TC-05: Reset ---
Write-Host "--- TC-05: Reset to Full-Screen ---" -ForegroundColor Yellow
& "$ScriptDir\Reset-WacomMapping.ps1"
Write-Host ">> Verify: stylus should now cover the full display"
Write-Host ""
Write-Host "=== Test run complete. Fill SPIKE-RESULTS.md with observations. ===" -ForegroundColor Cyan
```
