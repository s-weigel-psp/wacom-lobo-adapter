```powershell
#Requires -Version 5.1
<#
.SYNOPSIS
    Restricts the Wacom stylus to an arbitrary screen region by importing a modified preference XML.

.PARAMETER X
    Left edge of the mapping region in screen pixels (physical pixels).

.PARAMETER Y
    Top edge of the mapping region in screen pixels (physical pixels).

.PARAMETER Width
    Width of the mapping region in screen pixels (physical pixels).

.PARAMETER Height
    Height of the mapping region in screen pixels (physical pixels).

.EXAMPLE
    .\Set-WacomMapping.ps1 -X 0 -Y 0 -Width 960 -Height 1080
    # Restricts stylus to left half of a 1920×1080 display.

.NOTES
    Requires spike\baseline-local.xml to exist (exported in Plan 01-01 setup).
    Per D-01 (CONTEXT.md): modifies a clone of the baseline, never the baseline itself.
    Per D-02 (CONTEXT.md): uses Select-Xml XPath for element selection.
    PrefUtil path and import flag verified in Plan 01-01 (see test-log.md).
#>
[CmdletBinding()]
param(
    [Parameter(Mandatory)][int]$X,
    [Parameter(Mandatory)][int]$Y,
    [Parameter(Mandatory)][int]$Width,
    [Parameter(Mandatory)][int]$Height
)

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

# --- Paths ---
$ScriptDir    = $PSScriptRoot
$BaselinePath = Join-Path $ScriptDir 'baseline-local.xml'
$TempPath     = Join-Path $env:TEMP 'wacom-mapping-temp.xml'

# SUBSTITUTION POINT: PrefUtil path from 01-01-SUMMARY.md
$PrefUtilPath = '$PREFUTIL_PATH_VALUE'

# --- Precondition checks ---
Write-Host "[Set-WacomMapping] Checking prerequisites..."

if (-not (Test-Path $BaselinePath)) {
    Write-Error "baseline-local.xml not found at: $BaselinePath`nRun Plan 01-01 to export the baseline first."
    exit 1
}

if (-not (Test-Path $PrefUtilPath)) {
    Write-Error "PrefUtil not found at: $PrefUtilPath`nVerify the path in this script matches the test machine."
    exit 1
}

Write-Host "[Set-WacomMapping] Parameters: X=$X Y=$Y Width=$Width Height=$Height"
Write-Host "[Set-WacomMapping] Right=$($X + $Width) Bottom=$($Y + $Height)"

# --- Clone baseline (D-01) ---
Write-Host "[Set-WacomMapping] Cloning baseline-local.xml to temp path..."
[xml]$doc = Get-Content -Path $BaselinePath -Raw -Encoding UTF8

# --- Locate mapping element via XPath (D-02) ---
Write-Host "[Set-WacomMapping] Locating screen-mapping element via XPath..."

# SUBSTITUTION POINT: XPath and optional -Namespace from 01-01-SUMMARY.md
# If XML has no namespace:
#   $mappingNode = Select-Xml -Xml $doc -XPath '$XPATH_EXPRESSION' | Select-Object -First 1 -ExpandProperty Node
# If XML has a namespace (xmlns="some-uri"):
#   $mappingNode = Select-Xml -Xml $doc -XPath '$XPATH_EXPRESSION' -Namespace @{ wacom = 'namespace-uri-from-summary' } | Select-Object -First 1 -ExpandProperty Node
$mappingNode = Select-Xml -Xml $doc -XPath '$XPATH_EXPRESSION' $NAMESPACE_BLOCK |
               Select-Object -First 1 -ExpandProperty Node

if ($null -eq $mappingNode) {
    Write-Error "XPath '$XPATH_EXPRESSION' returned no nodes.`nVerify the XPath in this script matches the baseline XML structure."
    exit 1
}

Write-Host "[Set-WacomMapping] Found mapping node: $($mappingNode.Name)"

# --- Set coordinate attributes ---
# SUBSTITUTION POINT: Attribute names from 01-01-SUMMARY.md
# Left/Top/Right/Bottom form (if SUMMARY uses Right/Bottom):
$mappingNode.$COORD_LEFT_ATTR  = $X
$mappingNode.$COORD_TOP_ATTR   = $Y
$mappingNode.$COORD_RIGHT_ATTR = $X + $Width
$mappingNode.$COORD_BOTTOM_ATTR = $Y + $Height
#
# X/Y/Width/Height form (if SUMMARY uses Width/Height directly):
# $mappingNode.$COORD_X_ATTR      = $X
# $mappingNode.$COORD_Y_ATTR      = $Y
# $mappingNode.$COORD_WIDTH_ATTR  = $Width
# $mappingNode.$COORD_HEIGHT_ATTR = $Height
#
# Use whichever form matches SUMMARY findings; delete the other.

Write-Host "[Set-WacomMapping] Coordinates set in XML."

# --- Save modified XML ---
$doc.Save($TempPath)
Write-Host "[Set-WacomMapping] Modified XML saved to: $TempPath"

# --- Import via PrefUtil (Pattern 2: Wait-Process, not -Wait flag) ---
Write-Host "[Set-WacomMapping] Invoking PrefUtil to import modified preferences..."

$elapsed = Measure-Command {
    # SUBSTITUTION POINT: Import flag from 01-01-SUMMARY.md ('--import' or '/import')
    $proc = Start-Process -FilePath $PrefUtilPath `
                          -ArgumentList '$IMPORT_FLAG', $TempPath `
                          -PassThru
    $proc | Wait-Process
}

Write-Host "[Set-WacomMapping] PrefUtil exit code: $($proc.ExitCode)"
Write-Host "[Set-WacomMapping] PrefUtil completed in $([Math]::Round($elapsed.TotalMilliseconds)) ms"

if ($proc.ExitCode -ne 0) {
    Write-Warning "PrefUtil returned non-zero exit code: $($proc.ExitCode). Mapping may not have been applied."
    exit $proc.ExitCode
}

Write-Host "[Set-WacomMapping] Done. Stylus should now be restricted to X=$X Y=$Y Width=$Width Height=$Height."
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| PowerShell script → PrefUtil.exe | Script passes a temp XML file path as argument; PrefUtil reads and applies it |
| temp XML file → filesystem | Modified clone written to %TEMP%; contains user mapping preferences |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-02-01 | Tampering | wacom-mapping-temp.xml in %TEMP% | accept | Temp file is ephemeral and session-scoped; no secrets; overwritten each invocation; low-value target in a spike context |
| T-01-02-02 | Tampering | spike/baseline-local.xml | mitigate | Script reads baseline as a clone source, never writes to it; baseline integrity preserved by only ever importing it unmodified in Reset-WacomMapping.ps1 |
| T-01-02-03 | Denial of Service | PrefUtil.exe invocation | accept | Scripts validate PrefUtil path before invoking; non-zero exit code triggers warning and early exit |
| T-01-02-04 | Spoofing | $PrefUtilPath in script | accept | Spike runs on developer-controlled test machine; path is set from SUMMARY.md discovery; no untrusted input |
</threat_model>
