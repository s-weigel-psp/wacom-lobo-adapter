```go
// needsServiceRestart is set based on Task 1 Investigation B findings.
// true = restart WtabletServicePro after write; false = service hot-reloads.

// typedStr represents an XML element with a "type" attribute and string text content.
// The Wacom preference XML uses type="integer", type="bool", type="string", type="double"
// on every element — these must round-trip unchanged to preserve the file format.
type typedStr struct {
    Type  string `xml:"type,attr"`
    Value string `xml:",chardata"`
}

type xyzMap struct {
    Type   string    `xml:"type,attr"`
    Extent typedCoord `xml:"Extent"`
    Origin typedCoord `xml:"Origin"`
}

type typedCoord struct {
    Type string    `xml:"type,attr"`
    X    typedStr  `xml:"X"`
    Y    typedStr  `xml:"Y"`
    Z    typedStr  `xml:"Z"`
}

// screenAreaPartial models only the fields we modify inside <ScreenArea>.
// All other children (Ballistic, CursorMode, Locks, etc.) are preserved
// via the RawContent field using xml:",innerxml".
// IMPORTANT: This approach requires a two-pass decode: first unmarshal into a full
// struct preserving all children, then marshal back. If innerxml is used, the
// sibling elements must not overlap with the named fields.
//
// Full struct approach (safer): model all known children explicitly.
// This avoids any risk of innerxml clobbering named fields.
type screenArea struct {
    XMLName          xml.Name        `xml:"ScreenArea"`
    TypeAttr         string          `xml:"type,attr"`
    AreaType         typedStr        `xml:"AreaType"`
    Ballistic        typedStr        `xml:"Ballistic"`
    CursorMode       typedStr        `xml:"CursorMode"`
    Locks            typedStr        `xml:"Locks"`
    MouseHeight      typedStr        `xml:"MouseHeight"`
    MouseOrientation typedStr        `xml:"MouseOrientation"`
    MouseSpeed       typedStr        `xml:"MouseSpeed"`
    Options          typedStr        `xml:"Options"`
    ScreenOutputArea xyzMap          `xml:"ScreenOutputArea"`
    SensitivityX     typedStr        `xml:"SensitivityX"`
    SensitivityY     typedStr        `xml:"SensitivityY"`
    WhichMonitor     typedStr        `xml:"WhichMonitor"`
}

// inputScreenAreaArrayElement models one ArrayElement under InputScreenAreaArray.
// Only the ScreenArea child is modified; InputArea is preserved via innerxml.
type inputScreenAreaArrayElement struct {
    XMLName  xml.Name   `xml:"ArrayElement"`
    TypeAttr string     `xml:"type,attr"`
    AreaID   typedStr   `xml:"AreaID"`
    InputAreaRaw string `xml:",innerxml"` // NOT used — see note below
    ScreenArea   screenArea `xml:"ScreenArea"`
}
```

NOTE on InputArea: The full struct approach requires modelling InputArea too (it has many children).
The RECOMMENDED approach for this specific use case is to decode the entire document into a
deep struct tree matching the schema exactly as seen in baseline-reference.Export.wacomxs lines 47-109.
Alternatively, use a raw bytes approach: read the file, find and replace only the ScreenArea subtree
using xml.Decoder token stream, preserving everything else as-is. Choose whichever produces a
byte-for-byte identical file for all non-mapping fields when round-tripping the baseline.

```go
//go:build windows

package wacom

import (
    "fmt"
    "log/slog"
    "time"

    "golang.org/x/sys/windows/svc"
    "golang.org/x/sys/windows/svc/mgr"
)

const wacomServiceName = "WtabletServicePro"
const serviceStopTimeout = 10 * time.Second

// RestartWacomService stops WtabletServicePro and starts it again via Windows SCM.
// Returns nil on success, or a structured error map on failure.
// Called only when needsServiceRestart is true (set from Task 1 findings).
func RestartWacomService(logger *slog.Logger) map[string]interface{} {
    m, err := mgr.Connect()
    if err != nil {
        logger.Error("SCM connect failed", slog.String("error", err.Error()))
        return errMap("cannot connect to Windows SCM: "+err.Error(), "ERR_SERVICE_CONNECT")
    }
    defer m.Disconnect()

    s, err := m.OpenService(wacomServiceName)
    if err != nil {
        logger.Error("service open failed", slog.String("service", wacomServiceName), slog.String("error", err.Error()))
        return errMap(fmt.Sprintf("cannot open service %q: %v", wacomServiceName, err), "ERR_SERVICE_NOT_FOUND")
    }
    defer s.Close()

    if _, err := s.Control(svc.Stop); err != nil {
        logger.Error("service stop failed", slog.String("error", err.Error()))
        code := "ERR_SERVICE_RESTART_FAILED"
        if isAccessDenied(err) {
            code = "ERR_SERVICE_ACCESS_DENIED"
        }
        return errMap("failed to stop service: "+err.Error(), code)
    }

    // Poll until stopped (timeout serviceStopTimeout)
    deadline := time.Now().Add(serviceStopTimeout)
    stopped := false
    for time.Now().Before(deadline) {
        status, err := s.Query()
        if err != nil {
            break
        }
        if status.State == svc.Stopped {
            stopped = true
            break
        }
        time.Sleep(200 * time.Millisecond)
    }
    if !stopped {
        return errMap("service did not stop within 10s", "ERR_SERVICE_RESTART_TIMEOUT")
    }

    if err := s.Start(); err != nil {
        return errMap("failed to start service: "+err.Error(), "ERR_SERVICE_RESTART_FAILED")
    }

    logger.Info("WtabletServicePro restarted successfully")
    return nil
}

// isAccessDenied checks whether an error represents a Windows Access Denied condition.
func isAccessDenied(err error) bool {
    // Check for ERROR_ACCESS_DENIED (5) in the error chain
    return err != nil && (containsStr(err.Error(), "Access is denied") || containsStr(err.Error(), "access denied"))
}

func containsStr(s, substr string) bool {
    return len(s) >= len(substr) && (s == substr || len(s) > 0 && containsByte(s, substr))
}
// Use strings.Contains instead:
import "strings"
func isAccessDenied(err error) bool {
    return err != nil && strings.Contains(strings.ToLower(err.Error()), "access")
}
```
